def login_user(user)
  session = Session.new
  session.token = SecureRandom.hex
  session.ip_address = request.remote_ip
  session.user = user

  cookies[:auth_token] = {
    value: session.token,
    expires: Time.now + Session::TOKEN_EXPIRE_TIME,
    domain: :all,
  }

  session.save!
end
